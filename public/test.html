üìù Integration Guide

To integrate this chat widget into your website, add this script tag:

<script>
  // Configure the SDK BEFORE loading it
  window.ChatEngineConfig = {
    apiBase: 'http://localhost:3000',
    apiKey: 'your_tenant_api_key_here'
  };
</script>
<script src="http://localhost:3000/chat-engine.sdk.js"></script>

<!-- Alternative: Manual initialization -->
<script>
  // Or configure after loading and manually initialize
  window.ChatEngineSDK.config({
    apiBase: 'http://localhost:3000',
    apiKey: 'your_tenant_api_key_here'
  });
  window.ChatEngineSDK.init();
</script>
